      \begin{alertblock}{Introduction}
        \begin{itemize}
          \item Long-standing debate in Mandarin: 
            \textbf{are modals clause-embedding lexical verbs (\emph{bi-clausal}, \Next), or higher heads in the same clause (\emph{mono-clausal}, \NNext)?}

          \item Both \Next and \NNext are intended to mean `Lisi can come.'
            \begin{tcolorbox}
              \ex. \begingl
                % \glpreamble  //
                \gla \rightcomment{\scriptsize \citet{linFinitenessClausesRaising2011,linMultiplemodalConstructionsMandarin2012,zhangSentencefinalAspectParticles2019}}Lisi \nogloss{\I{[}VP {} } @  \nogloss{\I{[}\vmod{} {} } @ keyi @ \nogloss{]} \nogloss{\I{[}TP/CP {} } @ \nogloss{\gap{}} lai @ \nogloss{]].} //
                \glb L. can come //
              \endgl

              \ex.  \begingl
                % \glpreamble  //
                \gla \rightcomment{\scriptsize \citet{tsaiTopographyChineseModals2015,yipModalMovementLicensed2022,erlewineLowSentencefinalParticles2017}}Lisi \nogloss{\I{[}ModP \I{[}Mod {} } @ keyi @ \nogloss{]} \nogloss{\I{[}\emph{v}P {} } @ \nogloss{\gap{}} lai @ \nogloss{]].} //
                \glb L. can come //
              \endgl

            \end{tcolorbox}

          \item There is evidence for the clause-embedding view drawing on \citet{chappellVariationGrammaticalizationComplementizers2008,huangControlComplementsMandarin2018}, who analyze \emph{shuo} as a complementizer:
            \begin{tcolorbox}
              \ex. \begingl
                \gla Lisi keyi \nogloss{\I{[}CP \I{[}C } @ \textbf{shuo} @ \nogloss{]} xian shi-le zai zuo jueding @ \nogloss{]}. //
                \glb L. can \textsc{comp} first try-\textsc{pfv} then make decision //

      \begin{block}{Argument 1: Locality of association with focus-sensitive operators}
        \begin{itemize}

          \item \textbf{Modals pattern with CP-embedding verbs in blocking association with \opfs}.
          \item \opfs{} \emph{ye} `also' and \emph{dou} `even' allow backward association:
            \begin{tcolorbox}
              \ex. \begingl
                \gla  Lisi\F{} \textbf{dou} \nogloss{/} \textbf{ye} lai-le. //
                \glb  L. even also come-\textsc{pfv} //
                \glft `Lisi\F{} even/also came.' //
              \endgl

            \end{tcolorbox}

          \item \opfs{} can associate with very high positions in the clause, at least TopicP:
            \begin{tcolorbox}
              \ex. \label{ex:base_topic} \begingl

                \glb \textsc{lian} seafood                      \textsc{1sg} even         only like   eat cook cooked \textsc{de} dish //
                \glft `Even for seafood\F{}, I only like to eat cooked dishes.' //
              \endgl

            \end{tcolorbox}
          \item \textbf{Proposal:} \opfs{} covertly move to c-command the associate, contra \citet{erlewineMovementOutFocus2014a}. 
            \begin{tcolorbox}[title=\opfs{} movement,colback=red!10!white,opacityfill=1]
              \ex. \label{ex:foc_assoc} \tikzmarknode{target}{\opfs} \ldots{} [ \ldots{} XP\F{} \ldots{} \tikzmarknode{base}{\gap{}} \ldots{} ] \hfill {\scriptsize \citep{lahiriFocusNegativePolarity1998,crnicNonmonotonicityNPILicensing2014,liuVarietiesAlternativesMandarin2017}}
              \begin{tikzpicture}[overlay, remember picture]
                \draw[-Stealth, line width=1mm] (base.south) -- ++(0, -.8) -| (target.south);
              \end{tikzpicture}
              \vspace{2ex}

\begin{block}{Argument 2: Scope of focus-sensitive operators}
        \begin{itemize}
          \item \textbf{Presuppositions of \opfs{} project universally and epistemically from the scope of modals, like CP-embedding verbs.}
          % \item \Next presupposes the epistemic statement that Lisi is \textbf{expected} to write something else tomorrow; it is infelicitous where Lisi is only \textbf{allowed} to write something else.
            % which is correctly predicted to be grammatical by Test 1 (both the \opfs{} and its associate are on the same side of the modal; \cref{ex:loc_cons} is not violated).
           \begin{tcolorbox}[title={Universal epistemic presupposition projected from under \emph{xiwang} `hope'}]
              \ex. \begingl
                \gla Lisi xiwang \nogloss{\I{[}CP } @  Zhangsan\F{} \textbf{ye} lai @ \nogloss{].} //
                \glb L. hope Z. also come //
                \glft `Lisi hopes that Zhangsan\F{} will also come.' //
              \endgl % \\
              \eqmakebox[presup]{Presupposes: \ding{51}}Lisi \textbf{believes} someone else will come. \\
              \eqmakebox[presup][r]{\ding{55}}Lisi \textbf{hopes} someone else will come.

            \end{tcolorbox} 
            \begin{tcolorbox}[title={Universal epistemic presupposition projected from under \emph{keyi} `can'}]
              \ex. \begingl
                \gla Lisi keyi \nogloss{[} @ lunwen\F{} \textbf{ye} mingtian xie @ \nogloss{]}. //
                \glb L. can paper also tomorrow write //
                \glft `Lisi is allowed to also write the paper\F{} tomorrow.' //
              \endgl % \\
              \eqmakebox[presup]{Presupposes: \ding{51}}Lisi is \textbf{expected} to write something else tomorrow.\\
              \eqmakebox[presup][r]{\ding{55}}Lisi is \textbf{allowed} to write something else tomorrow.

            \end{tcolorbox}
          % \item \Last is just like \Next, which uncontroversially involves CP-embedding
            
          % \item Example \LLast necessarily presupposes the epistemic statement that Lisi is \emph{expected} to write something else tomorrow.
          \item \textbf{Explanation for \Last}: 1. \textbf{\emph{ye} is interpreted below \emph{keyi}}, and 2. \textbf{presuppositions project universally and epistemically} à la \citet{heimPresuppositionProjectionSemantics1992}.
          \item \Last is unlike the English mono-clausal \emph{can} construction \Next, and like the bi-clausal \emph{allow} construction \NNext with embedded \emph{also}:
            \begin{tcolorbox}[title={Existential deontic presupposition for English \emph{can}}]
              \ex. Lisi can \textbf{also} write their paper\F{} tomorrow.\\
              \eqmakebox[presup]{Presupposes: \ding{51}}Lisi is \textbf{allowed} to write something else tomorrow. \\
              \eqmakebox[presup][r]{\ding{55}}Lisi is \textbf{expected} to write something else tomorrow.

            \end{tcolorbox}
            \begin{tcolorbox}[title={Universal epistemic presupposition projected from under English \emph{allow}},after skip=0pt,bottom=0pt]
              \ex. Lisi is allowed to \textbf{also} write their paper\F{} tomorrow.\\
              \eqmakebox[presup]{Presupposes: \ding{51}}Lisi is \textbf{expected} to write something else tomorrow. \\
              \eqmakebox[presup][r]{\ding{55}}Lisi is \textbf{allowed} to write something else tomorrow.

            \end{tcolorbox}
          % \item \Last only has the deontic presupposition that Lisi is \textbf{allowed} to write something else tomorrow, not that Lisi is \textbf{expected} to do so.
          % \item The reading where \emph{ye} outscopes \emph{keyi}, merely presupposing that Lisi is \emph{allowed} to write something else tomorrow, is unavailable.
          % \item \LLast is infelicitous where only the deontic presupposition is met.
            % \item Again, modalized sentences pattern with clause-embedding verbs \Last.
          % \item \emph{Ye} must be interpreted under \emph{xiwang} in \Last.

        \end{itemize}

      \end{block}

      \begin{exampleblock}{Consequence: Modal movement}
          \begin{itemize}
            \item If modals embed CPs, then subjects \textbf{raise} across them.
            \item Modal-initialness is the \textbf{base} form, casting doubt on modal movement analyses of modal-initialness \citep{laiMovingHeadsSpecifiers2024,yipModalMovementLicensed2022}.
              \begin{tcolorbox}[after skip=0pt,bottom=0pt]
                \ex. \begingl
                  \gla keyi Zhangsan qu canjia bisai. //
                  \glb can Z. go take.part contest //
                  \glft `It is allowed that Zhangsan takes part in the contest.' //
                \endgl

              \end{tcolorbox}
          \end{itemize} 
        \end{exampleblock}

      \begin{block}{Argument 3: Distribution of focus-sensitive operators}
        \begin{itemize}
          \item \textbf{Modals allow \opfs{} to precede items that they otherwise cannot in the same clause.}
          \item \opfs{} \emph{ye} `also' and \emph{dou} `even' cannot precede fronted objects (TP-internal topics in Spec,\topobj,  \citealt{chenObscuredUniversalityMandarin2023}).
          % \item Both \Next and \NNext mean `Lisi\F{} also wrote the paper.'
            \begin{tcolorbox}[title={Spec,\topobj{} must precede \opfs{}}]
              \ex. \begingl
                \gla Lisi\F{} \nogloss{\{} @ \textbf{lunwen}\(_1\) \textbf{ye} @ \nogloss{/} \nogloss{*} @ \textbf{ye} \textbf{lunwen}\(_1\) @ \nogloss{\}}  xie-le \nogloss{\gap{1}}. //
                \glb L. paper also also paper write-\textsc{pfv} //
                \glft `Lisi\F{} also wrote the paper.' //
              \endgl

              % \ex. \begingl
              %   \gla \ljudge{*}Lisi\F{} \textbf{ye} \textbf{lunwen}\(_1\) xie-le \nogloss{\gap{1}}. //
              %   \glb L. also paper write-\textsc{pfv} //
              %   \glft Intended: `Lisi\F{}  did his paper' //
              % \endgl
              %

              \tcblower
              \ex. \label{ex:order_top_fs} \topobj{} \(\prec\) \opfs{} \(\prec\) V

            \end{tcolorbox}
          % \item Spec,\topobj{} must be higher than \opfs{} at base:
          %   \begin{tcolorbox}[colback=red!10!white]
          %     \ex. \label{ex:order_top_fs} \topobj{} \(\prec\) \opfs{} \(\prec\) V
          %
          %   \end{tcolorbox}
          \item This is a \textbf{clause-internal} ordering constraint; \opfs{} in the matrix can precede Spec,\topobj{} in the embedded CP:
            \begin{tcolorbox}[title={Ordering constraint \cref{ex:order_top_fs} holds within the same CP}]
           \ex. \begingl
             \gla Zhangsan\F{} \textbf{ye} xiwang \nogloss{\I{[}\textbf{CP} } @ Lisi \textbf{lunwen}\(_1\) mingtian xie \nogloss{\gap{1}} @ \nogloss{].} //
             \glb Z. also hope L. paper tomorrow write //
             \glft `Zhangsan\F{} also hopes that Lisi will write the paper tomorrow.' //
           \endgl 

           \tcblower
           \ex. \label{ex:hope_order}\I{[}\textbf{CP} \opfs{} \(\prec\) V\textsubscript{CP} \(\prec\) \I{[}\textbf{CP} \topobj{} \(\prec\) V  ]]

            \end{tcolorbox}
          % \item When there are modals, \opfs{} can precede fronted objects:
          % \item Ignoring the modal, \Next has the sequence in \NNext:
          \item Modals also allow \opfs{} above them to precede Spec,\topobj{} below them.
            \begin{tcolorbox}[title={Spec,\topobj{} can follow \opfs{} with modals}]
              \ex.
              \begingl
% \glpreamble Fronted objects //
                \gla Lisi\F{} \textbf{ye} \textbf{keyi} \textbf{lunwen}\(_1\) mingtian xie \nogloss{\gap{1}}. //
                \glb L. also can paper tomorrow write //
                \glft `Lisi\F{} is also allowed to write the paper tomorrow.' //
              \endgl

              \tcblower
              \ex. \opfs{} \(\prec\) \vmod{} \(\prec\) \topobj{} \(\prec\) V 

            \end{tcolorbox}

          % \item The sequence in \Last is in conflict with \cref{ex:order_top_fs}.
          \item \textbf{Explanation:} Modals also embed CPs.
            \begin{tcolorbox}[title={Modals embed CPs, generating \Last, just like \cref{ex:hope_order} },colback=red!10!white,opacityfill=1]
              \ex. \I{[}\textbf{CP} \opfs{} \(\prec\) \vmod{} \(\prec\) \I{[}\textbf{CP} \topobj{} \(\prec\) V  ]]

            \end{tcolorbox}

          % \item The sequence in \Last can then be rewritten as \Next:
            % \begin{tcolorbox}[title=\cref{ex:order_top_fs} and \cref{ex:paradox} reconciled via two separate clausal ordering domains,colback=red!10!white,opacityfill=1]
            %   \ex. \I{[}CP \opfs{} \(\prec\) \vmod{} \(\prec\) \I{[}CP \topobj{} \(\prec\) V  ]]
            %
            % \end{tcolorbox}

          % \item In each CP in \Last, \cref{ex:order_top_fs} is not violated.
          \item \textbf{Correct prediction:} If \opfs{} and Spec,\topobj{} on the same side of modal, \cref{ex:order_top_fs} holds.
          % \item If the object is fronted across the modal, it precedes \opfs{}:
            \begin{tcolorbox}[title={Both Spec,\topobj{} and \opfs{} outside modal scope},colbacklower=white]
              \ex. \begingl
                \gla Lisi\F{} \nogloss{\{} @  \textbf{lunwen}\(_1\) \textbf{ye} \nogloss{/} \nogloss{*} @ \textbf{ye} \textbf{lunwen}\(_1\)  @ \nogloss{\}} \textbf{keyi} mingtian xie \nogloss{\gap{1}}. //
                \glb L. paper also also paper can tomorrow write //
                \glft `Lisi\F{} is also allowed write the paper tomorrow.' //
              \endgl

              \tcblower

              \ex. \I{[}\textbf{CP} \topobj{} \(\prec\) \opfs{} \(\prec\) \vmod{} \(\prec\) \I{[}\textbf{CP} V  ]] \label{ex:order_matrix}

            \end{tcolorbox}

          % \item Second, if \topobj{} and \opfs{} are both in the scope of the modal, the object also precedes \opfs{}:
            \begin{tcolorbox}[title={Both Spec,\topobj{} and \opfs{} inside modal scope},colbacklower=white,after skip=0pt,bottom=0pt]
              \ex.
              \begingl
                \gla Lisi \textbf{keyi} \nogloss{\{} @  \textbf{lunwen}\(_{1, \mathrm{F}}\) \textbf{ye} @ \nogloss{/} \nogloss{*} @ \textbf{ye} \textbf{lunwen}\(_{1, \mathrm{F}}\) @ \nogloss{\}} mingtian xie \nogloss{\gap{1}}. //
                \glb L. can  paper also also  paper tomorrow write //
                \glft `Lisi is allowed to also write the paper\F{} tomorrow.' //
              \endgl

              \tcblower
              \ex. \I{[}\textbf{CP} \vmod{} \(\prec\) \I{[}\textbf{CP} \topobj{} \(\prec\) \opfs{} \(\prec\) V  ]] \label{ex:order_embedded}

            \end{tcolorbox}

          % \item Clearly, both \cref{ex:order_matrix} and \cref{ex:order_embedded} follow \cref{ex:order_top_fs}.
          % \item This means that modals essentially `reset' the clausal projections and therefore the sequence that \opfs{} and \topobj{} follow.
        \end{itemize}
        \end{block}
\begin{exampleblock}{Consequence: Implicational Complementation Hierarchy}
        \begin{itemize}
            \item \textbf{ICH} \citep{wurmbrandImplicationalUniversalComplementation2020}: semantic clause-types (\emph{Event}, \emph{Situation}, \emph{Proposition}) have increasing \emph{minimum sizes} (\emph{v}P, TP, CP), but not \emph{upper bound sizes}.
              \item Deontic modals take complements of semantic type \emph{Event}, but nevertheless employ CPs, constituting evidence for the ICH.
        \end{itemize}
